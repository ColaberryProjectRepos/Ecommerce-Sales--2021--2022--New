Total Sales = SUM(ecommerce_data[sales_per_order])

Sales Per Month = DIVIDE([Total Sales],DISTINCTCOUNT('Calendar'[Month]),0)+0

Total Sales LM = CALCULATE(SUM(ecommerce_data[sales_per_order]), DATEADD('Calendar'[Date],-1,MONTH))

Sales MoM% = DIVIDE([Total Sales]-[Total Sales LM],[Total Sales LM])

Total Profit = sum(ecommerce_data[profit_per_order])

GPM = DIVIDE([Total Profit],[Total Sales],0)

GPM LM = CALCULATE([GPM],DATEADD('Calendar'[Date],-1,MONTH))

GPM MOM% = DIVIDE(([GPM]- [GPM LM]),[GPM LM])

YoY Sales% = 

VAR CurrentYearSales = SUM('ecommerce_data'[sales_per_order])

VAR PreviousYearSales = CALCULATE(

    SUM('ecommerce_data'[sales_per_order]), 

    DATEADD('Calendar'[Date], -1, YEAR)

)

RETURN

    IF(ISBLANK(PreviousYearSales), BLANK(), DIVIDE(CurrentYearSales - PreviousYearSales, PreviousYearSales))
